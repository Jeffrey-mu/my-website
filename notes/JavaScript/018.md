## 示例代码


```js
// 1 创建WeakMap 对象
let wm = new WeakMap()
const key = {name: 'zs'}
const key2 = {name: 'zs'}
// 2 set 
wm.set(key, 'hello')
wm.set(key2, 'hello')
// 3 get
console.log(wm.get(key)); // hello
// 3 has
console.log(wm.has(key)); // true
// 4 delete
wm.delete(key)
console.log(wm.has(key)); // false
// 5 clear
wm.clear()
console.log(wm.has(key)); // false
console.log(wm.has(key2)); // false
```
