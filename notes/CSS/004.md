    <span class="active">选中</span>
</div>

.css003 .active{
    font-size: 13px;

    position: relative;
}

    content: " ";
    width: 24px;
    height: 2px;

    position: absolute;

	.active{
    	font-size: 18px;

         &:after {
             content: " ";
             width: 24px;
             height: 2px;
             background: #FFAA00;
             position: absolute;
             bottom: -0.px;
             left: 50%;

	}
